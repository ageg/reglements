\chapter{Affiliation à des regroupements d'associations}


\article{Affiliation}

\alinea{L'AGEG est membre des regroupements d'associations suivants:
\begin{itemize}

\item La Fédération Étudiante de l'Université de Sherbrooke (FEUS)

\item Le Regroupement étudiant de maîtrise, diplôme et doctorat de l'Université de Sherbrooke (REMDUS)

\item La Confédération pour le rayonnement étudiant en ingénierie au Québec (CRÉIQ)

\item La Fédération canadienne étudiante en génie (FCEG)

\end{itemize}
}

\alinea{Toute personne membre peut présenter en assemblée générale une proposition visant à ce que la corporation devienne ou cesse d'être membre d'un regroupement d'associations.}
