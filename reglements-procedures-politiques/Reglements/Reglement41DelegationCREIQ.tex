\reglement{Relatif à la représentation des VPEX à la CRÉIQ}

\preambule{L’objectif du présent règlement est de donner des lignes directrices aux vice-présidents aux affaires externes (actifs et passifs) afin d’assurer la meilleure efficacité possible aux congrès de la CRÉIQ.}

\partie{Dispositions générales}
\article{Mission des VPEX à la CRÉIQ}
\alinea{Permettre d’affirmer les valeurs de l’AGEG auprès des membres de la CRÉIQ;}
\alinea{Faciliter les communications avec les autres associations membres de la CRÉIQ  ainsi qu’avec les autres instances externes;}
\alinea{Rechercher des solutions aux problèmes récurrents de l’AGEG;}
\alinea{Promouvoir l’OIQ et le Réseau IQ auprès des étudiants de l’AGEG.}

\article{Vision des VPEX à la CRÉIQ}
\alinea{La vision des VPEX à la CRÉIQ est d’assurer la continuité et le bon fonctionnement des activités de la CRÉIQ, telles que les Jeux de Génie et la Compétition québécoise d’ingénierie.}

\article{Valeurs des VPEX}
\valeurs
{Ouverture}{Travailler de concert avec les instances relatives au génie}
{Engagement}{Représenter l’AGEG à tous les congrès de la CRÉIQ pour y faire entendre ses valeurs}
{Intégrité}{S’assurer de l’honnêteté et de la justesse des actions de la CRÉIQ}
{Fraternité}{Favoriser un sentiment d’appartenance entre les étudiants de génie à travers le Québec}

\article{Rôles et pouvoirs}
\sousarticle{AGEG}
\alinea{Elle détermine en CA si elle désire s’investir dans l’organisation d’un congrès régulier ou congrès annuel}
\alinea{Elle se questionne à savir si elle veut proposer un candidat à titre d'administrateur de la CRÉIQ et le cas échéant, lui signe une lettre d'appui}
\article{VPEX actif de l’AGEG}
\alinea{Il est le représentant officiel de l’AGEG à la CRÉIQ.}
\sousalinea{VPEX passif de l’AGEG}
\alinea{Il peut remplacer le VPEX actif comme administrateur à la CRÉIQ avec le consentement de celui-ci;}
\alinea{Il est assuré de pouvoir être présent au congrès.}

\adoption{XX juillet 2017}{27 juillet 2017}
